\text{Wehnelt Cylinder Voltage is fixed at } 5KV \newline
\text{Accelerating anode voltage can be varied from 0-10}KV \newline
\text{Hot cathode coil voltage can be varied from 0-10}V \newline
\text{Plate voltage can be varied from } {-1}KV \text{ to } 1KV \newline
\\[2em]

f(V_h)=
\begin{cases}
      0                                    & 0V \leq V_h \lt 2.5V   \\
      \frac{1}{7.5^2} \times (V_h - 2.5)^2 & 2.5V \leq V_h \leq 10V
\end{cases}
\\[2em]
g(V_a)=
\begin{cases}
      0                            & 0V  \leq V_a  \lt 1KV   \\
      \frac{1}{2} \times (V_a - 1) & 1KV \leq  V_a \lt 3KV   \\
      1                            & 3KV \leq  V_a \leq 10KV \\
\end{cases}
\\[2em]

\text{Beam Intensity: } \newline
I(V_h, V_a)=f(V_h) \times g(V_a)
\\[2em]

\text{Beam deflection } y \text{ between plates at distance } x \text{: } \newline
y=0.25 \times \frac{V_p}{V_a} \times 0.1 \times x^2
